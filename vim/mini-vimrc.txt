"" BASIC
set nocompatible	          " no vi compatibility
filetype plugin indent on   " activates filetype specific features
set autoread		            " auto read when a file is changed from the outside
set nobackup		            " get rid of anoying ~file
set encoding=utf-8	        " set encoding to UTF-8
let mapleader=","	          " set mapleader to ,
set mouse=a                 " activate mouse for scrolling,...

"" LOOK
syntax on		                " syntax highlighting
set number		              " line numbers
set relativenumber	        " all lines show their relative number, exept for current line
set laststatus=2	          " always show statusbar
set ruler                   " always show position
set wildmenu		            " enable enhanced command-like completion
set background=dark	        " set background dark
colorscheme desert	        " colorscheme desert

"" SEARCH
set ignorecase		          " ignore case when searching
set smartcase		            " when searching try to be smart about cases
set hlsearch		            " highlight search results
set showmatch		            " show matching brackets when text indicator is over them

"" INDENTING
set expandtab		            " insert space characters whenever the tab key is pressed
set shiftwidth=2	          " size of indentation 
set tabstop=2   	          " number of spaces for a TAB
set autoindent		          " auto indenting

"" MAPPING
" Splits
map <C-h> <C-W>h	          " shortcuts for better split navigation
map <C-j> <C-W>j
map <C-k> <C-W>k
map <C-l> <C-W>l
" Tabs
nnoremap tg :tabnew<Space>  " shortcuts for tab generation
nnoremap tn :tabnext<Space>
nnoremap tp :tabprev<Space>
nnoremap tf :tabfirst<Space>
nnoremap tl :tablast<Space>
nnoremap tc :tabclose<Space>
" Fast saving
map <leader>w :w!<cr>
" Better search
map <space> /
